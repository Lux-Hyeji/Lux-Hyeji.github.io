# p.93 
n, m, k = map(int,input().split())

number = list(map(int,input().split()))
sum = 0

number.sort()    

while m > 0:

    for _ in range(k):
        sum += number[-1]

    if m - k > 0:
        sum += number[-2]
    elif m - k == 0:
        break
    m -= (k+1)

print(sum)
